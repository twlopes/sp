jQuery • highlightTextarea • Demos
Change highlighting color
$("textarea").highlightTextarea({
    words: ["Lorem ipsum","vulputate"],
    color: "#ADF0FF"
});
Case insensitive
$("textarea").highlightTextarea({
    words: ["lOrEm IpSuM","VULPUTATE"],
    caseSensitive: false
});
Resizing (with jQueryUI)
$("textarea").highlightTextarea({
    words: ["Lorem ipsum","Nam varius","Nullam"],
    resizable: true
});
Use regular expressions
$("textarea").highlightTextarea({
    words: ["(([0-9]{2})([\.]?)){5}"]
});
Customise highlight element
$("textarea").highlightTextarea({
    words: ["Donec tincidunt","vulputate"],
    color: "#99f",
    id: "demo-wrap"
});
#demo-wrap .highlight {
    border-radius: 0.5em;
    padding: 0 3px;
    margin: 0 -3px;
}
